#include <stdio.h>

const char* checkBalance(int arr[], int n) {
    long long s = 0, p = 0;
    for (int i = 0; i < n; i++) {
        s += arr[i];
    }
    for (int i = 0; i < n; i++) {
        long long suf = s - p - arr[i];
        if (p == suf) {
            return "YES";
        }
        p += arr[i];
    }
    return "NO";
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int l;
        scanf("%d", &l);
        int a[l];
        for (int j = 0; j < l; j++) {
            scanf("%d", &a[j]);
        }
        printf("%s\n", checkBalance(a, l));
    }
    return 0;
}
