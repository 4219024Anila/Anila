#include <stdio.h>
#include <stdlib.h>

void cs(int x, int* y) {
    int* z = calloc(100, sizeof(int));
    if (!z) return;

    for (int a = 0; a < x; a++) {
        z[y[a]]++;
    }

    for (int b = 0; b < 100; b++) {
        printf("%d ", z[b]);
    }
    printf("\n");

    free(z);
}
int main() {
    int p;
    scanf("%d", &p);

    int* q = malloc(p * sizeof(int));
    if (!q) return 1;

    for (int i = 0; i < p; i++) {
        scanf("%d", &q[i]);
    }

    cs(p, q);
    free(q);
    return 0;
}
