#include <stdio.h>
#include <stdlib.h>

int diagonalDiff(int size, int** matrix) {
    int sumMain = 0, sumAnti = 0;

    for (int k = 0; k < size; k++) {
        sumMain += matrix[k][k];
        sumAnti += matrix[k][size - 1 - k];
    }

    return abs(sumMain - sumAnti);
}

int main() {
    int n;
    scanf("%d", &n);

    int** mat = malloc(n * sizeof(int*));
    for (int i = 0; i < n; i++) {
        mat[i] = malloc(n * sizeof(int));
        for (int j = 0; j < n; j++) {
            scanf("%d", &mat[i][j]);
        }
    }

    printf("%d\n", diagonalDiff(n, mat));

    for (int i = 0; i < n; i++) {
        free(mat[i]);
    }
    free(mat);

    return 0;
}
