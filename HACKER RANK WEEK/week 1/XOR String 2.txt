#include <stdio.h>
#include <string.h>

void strings_xor(const char *s, const char *t, char *res) {
    int len = strlen(s);
    for (int i = 0; i < len; i++) {
        // XOR the characters directly
        if (s[i] == t[i])
            res[i] = '0';
        else
            res[i] = '1';
    }
    res[len] = '\0'; // Null-terminate the result string
}

int main() {
    char s[1001], t[1001], res[1001];

    // Read two binary strings
    scanf("%s", s);
    scanf("%s", t);

    strings_xor(s, t, res);
    printf("%s\n", res);

    return 0;
}
