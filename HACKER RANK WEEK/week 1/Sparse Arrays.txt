#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_STRINGS 1000
#define MAX_LENGTH 105

// Struct to hold string and its count
typedef struct {
    char str[MAX_LENGTH];
    int count;
} StringCount;

int main() {
    int n;
    scanf("%d", &n);

    StringCount list[MAX_STRINGS];
    int uniqueCount = 0;

    // Read n strings and count frequencies
    for (int i = 0; i < n; ++i) {
        char temp[MAX_LENGTH];
        scanf("%s", temp);

        int found = 0;
        for (int j = 0; j < uniqueCount; ++j) {
            if (strcmp(list[j].str, temp) == 0) {
                list[j].count++;
                found = 1;
                break;
            }
        }

        if (!found) {
            strcpy(list[uniqueCount].str, temp);
            list[uniqueCount].count = 1;
            uniqueCount++;
        }
    }

    int q;
    scanf("%d", &q);

    // Handle queries
    for (int i = 0; i < q; ++i) {
        char query[MAX_LENGTH];
        scanf("%s", query);

        int found = 0;
        for (int j = 0; j < uniqueCount; ++j) {
            if (strcmp(list[j].str, query) == 0) {
                printf("%d\n", list[j].count);
                found = 1;
                break;
            }
        }

        if (!found) {
            printf("0\n");
        }
    }

    return 0;
}
