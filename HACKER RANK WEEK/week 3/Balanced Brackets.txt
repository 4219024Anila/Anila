#include <stdio.h>
#include <string.h>

#define MAX_LEN 1000

int isMatchingPair(char open, char close) {
    return (open == '(' && close == ')') ||
           (open == '{' && close == '}') ||
           (open == '[' && close == ']');
}

int isBalanced(char* line) {
    char stack[MAX_LEN];
    int top = -1;

    for (int i = 0; line[i] != '\0' && line[i] != '\n'; i++) {
        char ch = line[i];

        if (ch == '(' || ch == '{' || ch == '[') {
            stack[++top] = ch;
        } else if (ch == ')' || ch == '}' || ch == ']') {
            if (top == -1 || !isMatchingPair(stack[top], ch)) {
                return 0;  // Not balanced
            }
            top--;  // Pop from stack
        }
    }

    return top == -1;  // Stack should be empty if balanced
}

int main() {
    int t;
    scanf("%d", &t);
    getchar(); // Consume the newline after the number

    while (t--) {
        char line[MAX_LEN];
        fgets(line, MAX_LEN, stdin);

        if (isBalanced(line)) {
            printf("YES\n");
        } else {
            printf("NO\n");
        }
    }

    return 0;
}
