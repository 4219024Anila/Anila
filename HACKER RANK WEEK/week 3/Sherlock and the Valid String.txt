#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int compare(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int main() {
    char str[100001];
    int arr[26] = {0};
    int i, len, l = 0, count = 0;

    scanf("%s", str);
    len = strlen(str);

    // Count character frequencies
    for(i = 0; i < len; i++) {
        arr[str[i] - 'a']++;
    }

    // Sort frequency array
    qsort(arr, 26, sizeof(int), compare);

    // Skip leading zeroes
    for(i = 0; i < 26; i++) {
        if(arr[i] != 0) {
            l = i;
            break;
        }
    }

    // Check validity
    if(arr[l] == 1) {
        count++;
        l++;
    }

    for(i = l + 1; i < 26; i++) {
        if(arr[i] != arr[l]) {
            count++;
        }
    }

    if(count > 1)
        printf("NO\n");
    else
        printf("YES\n");

    return 0;
}
